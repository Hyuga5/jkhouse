# ── Add this to your site's .htaccess (Apache / cPanel hosting) ──
# This makes /product/<anything> serve the same product.html file,
# the same way /product already works today. Without this, typing
# or refreshing a URL like /product/cpvcpipe directly will 404 —
# it will only work when reached by clicking inside the app.

<IfModule mod_rewrite.c>
  RewriteEngine On

  # If you don't already have a rule mapping /product -> product.html,
  # add this too:
  RewriteRule ^product/?$ /product.html [L]

  # New: anything under /product/ (e.g. /product/cpvcpipe) -> product.html
  RewriteCond %{REQUEST_FILENAME} !-f
  RewriteCond %{REQUEST_FILENAME} !-d
  RewriteRule ^product/[^/]+/?$ /product.html [L]
</IfModule>
